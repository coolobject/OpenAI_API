python -m venv myenv

# 2. 가상환경 구동
.\myenv\Scripts\activate

# python -m pip install --upgrade pip

pip install openai python-dotenv ipykernel

# 5. 커널 등록해두기 (선택)
python -m ipykernel install --user --name=myenv
